 479    private GameObject FindNearestEnemy()
 480    {
 481        // Find all potential targets within extended range
 482        Collider2D[] colliders = Physics2D.OverlapCircleAll(transform.position, attackRange * 1.5f);
 483        GameObject nearestEnemy = null;
 484        float nearestDistance = float.MaxValue;
 485
 486        foreach (Collider2D collider in colliders)
 487        {
 488            // Check for regular enemies with EnemyBehavior
 489            EnemyBehavior enemyBehavior = collider.GetComponent<EnemyBehavior>();
 490            DragonBoss dragonBoss = collider.GetComponent<DragonBoss>();
 491
 492            // Target regular enemies OR the dragon boss
 493            bool isValidTarget = false;
 494            bool isDead = false;
 495
 496            if (enemyBehavior != null)
 497            {
 498                isValidTarget = true;
 499                isDead = enemyBehavior.IsDead;
 500            }
 501            else if (dragonBoss != null)
 502            {
 503                isValidTarget = true;
 504                isDead = dragonBoss.IsDead; // Assuming DragonBoss has an IsDead property
 505            }
 506
 507            if (isValidTarget && !isDead)
 508            {
 509                // Extra safety check - don't attack player or other summons
 510                if (collider.gameObject.tag != "Player" && collider.gameObject.tag != "PlayerSummon")
 511                {
 512                    float distance = Vector3.Distance(transform.position, collider.transform.position);
 513                    if (distance < nearestDistance)
 514                    {
 515                        nearestDistance = distance;
 516                        nearestEnemy = collider.gameObject;
 517                    }
 518                }
 519            }
 520        }
 521
 522        return nearestEnemy;
 523    }
